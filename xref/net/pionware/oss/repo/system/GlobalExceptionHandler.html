8   import com.fasterxml.jackson.core.JsonProcessingException;
9   import com.fasterxml.jackson.databind.ObjectMapper;
10  
11  import net.pionware.oss.repo.exception.GitHubErrorException;
12  import net.pionware.oss.repo.model.error.ServerError;
13  
14  @ControllerAdvice
15  public class GlobalExceptionHandler {
16  
17  	/*
18  	 * This type of exception contains a message (string) in json format.
19  	 * Essentially this can be forwarded "as is" to frontend which should be able to
20  	 * handle the json effectively
21  	 */
22      @ExceptionHandler(GitHubErrorException.class)
23      public ResponseEntity<String> handleServerErrorException(GitHubErrorException ex) {
24      	
25          return new ResponseEntity<>(ex.getMessage(), HttpStatus.INTERNAL_SERVER_ERROR);
26      }
27      
28      /*
29       * This type of exception (any type) is converted to json so as
30       * the frontend can handle all the exceptions almost in a similar way 
31       */
32      @ExceptionHandler(Throwable.class)
33      public ResponseEntity<String> handleException(Exception ex) {
34      	String json = "{\"status\":500,\"message\":\"Internal Server Error\"}";
35      	
36          try {
37          	ServerError error = new ServerError(500, ex.getMessage());
38              ObjectMapper objectMapper = new ObjectMapper();
39              json = objectMapper.writeValueAsString(error);
40          } catch (JsonProcessingException e) {
41              
42          }
43          return new ResponseEntity<>(json, HttpStatus.INTERNAL_SERVER_ERROR);
44      }
45  }
