15  @Service
16  public class RunServiceImpl implements RunService {
17  	private static final Logger LOGGER = LogManager.getLogger(RunServiceImpl.class);
18  	
19      @Autowired
20      private RunItemRepository runItemRepository;
21      @Autowired
22      private SettingService settingService;
23      @Autowired
24      private RepoService repoService;
25      
26      @Override
27      public List<RunItemEntity> getAllRunItems() {
28          return runItemRepository.findAll();
29      }
30  
31      @Override
32      public RunItemEntity getRunItemById(Long id) {
33      	RunItemEntity item = runItemRepository.findById(id).orElse(null);
34      	validateRunItem(item);
35          
36          return item;
37      }
38      
39      @Override
40      public RunItemEntity saveRunItem(RunItemEntity runItem) {
41  
42          String maxLimitValue = settingService.getSettingByKey("MAX_LIMIT").getValue();
43          runItem.setMaxLimit(Long.parseLong(maxLimitValue));
44  
45          runItem.setCreatedAt(LocalDateTime.now());
46          runItem.setStatus("INITIALIZED");
47          
48          return runItemRepository.save(runItem);
49      }
50  
51      @Override
52      public RunItemEntity updateRunItem(RunItemEntity runItem) {
53      	RunItemEntity item = getRunItemById(runItem.getId());
54      	validateRunItem(item);
55      	
56          return runItemRepository.save(runItem);
57     }
58      
59      @Override
60      public void deleteRunItem(Long id) {
61          RunItemEntity runItem = getRunItemById(id);
62          validateRunItem(runItem);
63          
64          String clonePath = settingService.getSettingByKey("CLONE_PATH").getValue();
65       
66          File folder = new File(clonePath, id.toString());
67          boolean deleted = deleteDirectory(folder);
68          if(LOGGER.isInfoEnabled()) {
69          	LOGGER.info("Folder {} was deleted: {}", folder.getName(), deleted);
70          }
71    
72          if (!deleted && folder.exists()) {
73          	throw new IllegalStateException("Failed to delete directory: " + folder.getAbsolutePath());
74          }
75  
76          repoService.deleteRepositories(runItem);
77          
78          runItemRepository.delete(runItem);
79      }
80      
81      public boolean deleteDirectory(File directory) {
82          if (directory.isDirectory()) {
83              File[] files = directory.listFiles();
84              if (files != null) {
85                  for (File file : files) {
86                      if (!deleteDirectory(file)) {
87                          return false;
88                      }
89                  }
90              }
91          }
92          return directory.delete();
93      }
94      
95      @Override
96      public RunItemEntity initCloning(Long id) {
97          RunItemEntity runItem = getRunItemById(id);
98          validateRunItem(runItem);
99  
100         String clonePath = settingService.getSettingByKey("CLONE_PATH").getValue();
101      
102         File folder = new File(clonePath, id.toString());
103         boolean created = folder.mkdirs();
104         if(LOGGER.isInfoEnabled()) {
105         	LOGGER.info("Folder {} was created: {}", folder.getName(), created);
106         }
107   
108         if (!created && folder.exists()) {
109         	throw new IllegalStateException("Failed to create directory: " + folder.getAbsolutePath());
110         }
111             
112         runItem.setStatus("SELECTION_STARTED");
113         runItemRepository.save(runItem);
114      
115         return runItem;
116     }
117     
118 
119     private void validateRunItem(RunItemEntity runItem) {
120         if(runItem == null) {
121         	throw new IllegalStateException("Failed to find the run item");
122         }
123     }
