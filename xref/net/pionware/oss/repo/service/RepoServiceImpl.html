140     public Map<Integer, List<RepoItemEntity>> getSample(Long runId) {
141         // Step 1: Fetch repositories from the database for the given run ID
142         List<RepoItemEntity> repoItems = repoItemRepository.findByRunItemIdOrderByCreatedAtAsc(runId);
143 
144         // Step 2: Filter by MIT and Apache-2.0 licenses
145         List<RepoItemEntity> filteredRepoItems = filterByLicense(repoItems);
146 
147         // Step 3: Group repositories by creation year
148         Map<Integer, List<RepoItemEntity>> groupedRepoItems = groupByCreationYear(filteredRepoItems);
149 
150         // Step 4: Shuffle and select samples per year
151         int maxNumber = Integer.parseInt(settingService.getSettingByKey("SAMPLE_NUM").getValue());
152         if(LOGGER.isInfoEnabled()) {
153         	LOGGER.info("Sampling {} repositories per year", maxNumber);
154         }
155         Map<Integer, List<RepoItemEntity>> sampledRepos = shuffleRepositories(groupedRepoItems, maxNumber);
156 
157         // Step 5: Save sampled repositories to the database
158         saveSampledRepositories(sampledRepos);
159      
160         // Step 6: update status
161         RunItemEntity runItem = runItemRepository.findById(runId).get();
162         runItem.setStatus("ANALYIS_STARTED");
163         runItemRepository.save(runItem);
164         
165         return sampledRepos;
166     }

168     // Filters repositories based on accepted licenses (MIT, Apache-2.0)
169     private List<RepoItemEntity> filterByLicense(List<RepoItemEntity> repoItems) {
170        // Set<String> acceptedLicenses = Set.of("", "other");
171         return repoItems.stream()
172             .filter(repo -> repo.getLicense() != null && !repo.getLicense().isEmpty() && !repo.getLicense().equalsIgnoreCase("other"))
173             .collect(Collectors.toList());
174     }

176     // Groups repositories by creation year
177     private Map<Integer, List<RepoItemEntity>> groupByCreationYear(List<RepoItemEntity> repoItems) {
178         Map<Integer, List<RepoItemEntity>> grouped = new HashMap<>();
179         for (RepoItemEntity repo : repoItems) {
180             int year = LocalDateTime.ofInstant(repo.getCreatedAt(), ZoneId.of("UTC")).getYear();
181             grouped.putIfAbsent(year, new ArrayList<>());
182             grouped.get(year).add(repo);
183         }
184         return grouped;
185     }

187     // Shuffles repositories and selects up to the target number per year
188     private Map<Integer, List<RepoItemEntity>> shuffleRepositories(Map<Integer, List<RepoItemEntity>> groupedRepos, int expectedTotal) {
189         Map<Integer, List<RepoItemEntity>> sampledRepositories = new HashMap<>();
190         Random random = new Random();
191 
192         // Calculate total number of available repositories across all years
193         int totalAvailable = groupedRepos.values().stream().mapToInt(List::size).sum();
194         if(LOGGER.isInfoEnabled()) {
195         	LOGGER.info("Total repositories available for sampling: {}", totalAvailable);
196         }
197         // Avoid division by zero if there are no repositories
198         if (totalAvailable == 0) {
199             return sampledRepositories;
200         }
201 
202         // Calculate proportional allocation for each year
203         Map<Integer, Integer> allocationPerYear = new HashMap<>();
204         for (Map.Entry<Integer, List<RepoItemEntity>> entry : groupedRepos.entrySet()) {
205             int proportionalCount = (int) Math.round(((double) entry.getValue().size() / totalAvailable) * expectedTotal);
206             if(LOGGER.isInfoEnabled()) {
207             	LOGGER.info("Proportional count for year {}: {} ", entry.getKey(), proportionalCount);
208             }
209             allocationPerYear.put(entry.getKey(), proportionalCount);
210         }
211 
212         // Adjust to ensure the total does not exceed expectedTotal
213         int actualTotal = allocationPerYear.values().stream().mapToInt(Integer::intValue).sum();
214         if (actualTotal > expectedTotal) {
215             List<Integer> keys = new ArrayList<>(allocationPerYear.keySet());
216             while (actualTotal > expectedTotal) {
217                 int key = keys.get(random.nextInt(keys.size()));
218                 if (allocationPerYear.get(key) > 0) {
219                     allocationPerYear.put(key, allocationPerYear.get(key) - 1);
220                     actualTotal--;
221                 }
222             }
223         }
224 
225         // Shuffle and select repositories based on proportional allocation
226         for (Map.Entry<Integer, List<RepoItemEntity>> entry : groupedRepos.entrySet()) {
227             List<RepoItemEntity> repos = new ArrayList<>(entry.getValue());
228             Collections.shuffle(repos, random);
229 
230             int numToSelect = allocationPerYear.getOrDefault(entry.getKey(), 0);
231             if(LOGGER.isInfoEnabled()) {
232             	LOGGER.info("{} selected repositories for year {} ", numToSelect, entry.getKey());
233             }
234             List<RepoItemEntity> selected = repos.stream().limit(numToSelect).collect(Collectors.toList());
235 
236             sampledRepositories.put(entry.getKey(), selected);
237         }
238 
239         return sampledRepositories;
240     }

242     @Transactional
243     public void saveSampledRepositories(Map<Integer, List<RepoItemEntity>> sampledRepositories) {
244     	List<RepoItemEntity> entities = sampledRepositories.values().stream()
245                 .flatMap(List::stream)
246                 .map(repo -> {
247                 	repo.setChosen(true);
248                 	return repo;
249                 })
250                 .collect(Collectors.toList());
251 
252             repoItemRepository.saveAll(entities);
253     }
